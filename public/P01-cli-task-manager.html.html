Quick intro to structs

A struct is a custom type that groups related fields together. Think of it as a blueprint for an object. Our Task struct has three fields: an ID (int), a Text description (string), and a Done flag (bool). We'll go deep on structs in M07 — for now, just know that Task{ID: 1, Text: "Learn Go", Done: false} creates one.

§3 Build It Step by Step

We're going to build this incrementally — one function at a time. After each step, run the program to verify it works before moving on. This is how real software is built.

Step 1 — printUsage()

The first function we write is the help text. Every good CLI tool tells you how to use it. Add this after the nextID variable:

Go main.go step 1 
func printUsage() {
    fmt.Println("Usage: tasks [command]")
    fmt.Println()
    fmt.Println("Commands:")
    fmt.Println("  add <text>    Add a new task")
    fmt.Println("  list          List all tasks")
    fmt.Println("  done <id>    Mark a task as complete")
    fmt.Println("  delete <id>  Delete a task")
}

Step 2 — addTask()

Now let's add the ability to create a task. We create a new Task struct, append it to our global tasks slice, and increment the ID counter:

Go main.go step 2 
func addTask(text string) {
    if text == "" {
        fmt.Println("❌ Task text cannot be empty.")
        return
    }

    task := Task{
        ID:   nextID,
        Text: text,
        Done: false,
    }

    tasks = append(tasks, task)
    fmt.Printf("✅ Added task #%d: %q\n", nextID, text)
    nextID++
}
Go addTask — annotated
if text == "" { ... }
Guard clause — validate input first, bail out early if bad. Never let bad data into your data store.
task := Task{ ID: nextID, Text: text, Done: false }
Struct literal — create a new Task value with named fields. Any field you don't set gets its zero value.
tasks = append(tasks, task)
Append to the global slice. Remember: always reassign the result of append!
%q in Printf
The %q verb prints a string with surrounding quotes. Great for showing the user exactly what text was saved.
nextID++
Increment after use. The next task will get ID 2, then 3, and so on.

Step 3 — listTasks()

Display all tasks. We loop over the slice with range and print a different symbol depending on whether each task is done:

Go main.go step 3 
func listTasks() {
    if len(tasks) == 0 {
        fmt.Println("📭 No tasks yet. Add one with: add <text>")
        return
    }

    fmt.Println("📋 Your Tasks:")
    for _, task := range tasks {
        status := "[ ]"
        if task.Done {
            status = "[✓]"
        }
        fmt.Printf("   %s %d. %s\n", status, task.ID, task.Text)
    }
}

Step 4 — findTask() Helper

Both markDone and deleteTask need to find a task by ID. Instead of writing the same search logic twice, we extract it into a helper. This is good engineering — Don't Repeat Yourself:

Go main.go step 4 
// findTask returns the index of a task by ID, or -1 if not found
func findTask(id int) int {
    for i, task := range tasks {
        if task.ID == id {
            return i
        }
    }
    return -1  // sentinel value: -1 means not found
}

Step 5 — markDone()

Go main.go step 5 
func markDone(id int) {
    idx := findTask(id)
    if idx == -1 {
        fmt.Printf("❌ No task with ID #%d found.\n", id)
        return
    }
    if tasks[idx].Done {
        fmt.Printf("⚠️  Task #%d is already done!\n", id)
        return
    }
    tasks[idx].Done = true
    fmt.Printf("🎉 Task #%d marked as complete!\n", id)
}

Step 6 — deleteTask()

Here's our M06 slice-delete idiom in real action — joining everything before and after the target index:

Go main.go step 6 
func deleteTask(id int) {
    idx := findTask(id)
    if idx == -1 {
        fmt.Printf("❌ No task with ID #%d found.\n", id)
        return
    }

    // The slice delete pattern from M06:
    // join everything before idx with everything after idx
    tasks = append(tasks[:idx], tasks[idx+1:]...)
    fmt.Printf("🗑️  Task #%d deleted.\n", id)
}

Step 7 — Wire it all up in main()

Now the final piece — the command router. os.Args gives us the command-line arguments as a slice of strings. os.Args[0] is always the program name itself, so real arguments start at index 1:

Go main.go step 7 — the final piece 
func main() {
    // os.Args[0] = program name, os.Args[1] = first argument
    if len(os.Args) < 2 {
        printUsage()
        return
    }

    command := os.Args[1]

    switch command {

    case "add":
        if len(os.Args) < 3 {
            fmt.Println("❌ Usage: add <task text>")
            return
        }
        addTask(os.Args[2])

    case "list":
        listTasks()

    case "done":
        if len(os.Args) < 3 {
            fmt.Println("❌ Usage: done <id>")
            return
        }
        id, err := strconv.Atoi(os.Args[2])
        if err != nil {
            fmt.Printf("❌ Invalid ID: %q — must be a number\n", os.Args[2])
            return
        }
        markDone(id)

    case "delete":
        if len(os.Args) < 3 {
            fmt.Println("❌ Usage: delete <id>")
            return
        }
        id, err := strconv.Atoi(os.Args[2])
        if err != nil {
            fmt.Printf("❌ Invalid ID: %q — must be a number\n", os.Args[2])
            return
        }
        deleteTask(id)

    default:
        fmt.Printf("❌ Unknown command: %q\n", command)
        fmt.Println()
        printUsage()
    }
}
👀
strconv.Atoi — String to Integer

strconv.Atoi converts a string like "3" to an integer 3. It returns the int and an error — exactly the multiple return pattern from M05. If the user passes "abc" instead of a number, Atoi returns an error and we catch it cleanly.

§4 The Complete File

Here's the entire main.go assembled — verify yours matches before running:

Go main.go — complete file 
package main

import (
    "fmt"
    "os"
    "strconv"
)

type Task struct {
    ID   int
    Text string
    Done bool
}

var tasks []Task
var nextID = 1

func printUsage() {
    fmt.Println("Usage: tasks [command]")
    fmt.Println()
    fmt.Println("Commands:")
    fmt.Println("  add <text>    Add a new task")
    fmt.Println("  list          List all tasks")
    fmt.Println("  done <id>    Mark a task as complete")
    fmt.Println("  delete <id>  Delete a task")
}

func addTask(text string) {
    if text == "" {
        fmt.Println("❌ Task text cannot be empty.")
        return
    }
    task := Task{ID: nextID, Text: text, Done: false}
    tasks = append(tasks, task)
    fmt.Printf("✅ Added task #%d: %q\n", nextID, text)
    nextID++
}

func listTasks() {
    if len(tasks) == 0 {
        fmt.Println("📭 No tasks yet. Add one with: add <text>")
        return
    }
    fmt.Println("📋 Your Tasks:")
    for _, task := range tasks {
        status := "[ ]"
        if task.Done { status = "[✓]" }
        fmt.Printf("   %s %d. %s\n", status, task.ID, task.Text)
    }
}

func findTask(id int) int {
    for i, task := range tasks {
        if task.ID == id { return i }
    }
    return -1
}

func markDone(id int) {
    idx := findTask(id)
    if idx == -1 {
        fmt.Printf("❌ No task with ID #%d found.\n", id)
        return
    }
    if tasks[idx].Done {
        fmt.Printf("⚠️  Task #%d is already done!\n", id)
        return
    }
    tasks[idx].Done = true
    fmt.Printf("🎉 Task #%d marked as complete!\n", id)
}

func deleteTask(id int) {
    idx := findTask(id)
    if idx == -1 {
        fmt.Printf("❌ No task with ID #%d found.\n", id)
        return
    }
    tasks = append(tasks[:idx], tasks[idx+1:]...)
    fmt.Printf("🗑️  Task #%d deleted.\n", id)
}

func main() {
    if len(os.Args) < 2 {
        printUsage()
        return
    }
    command := os.Args[1]
    switch command {
    case "add":
        if len(os.Args) < 3 { fmt.Println("❌ Usage: add <task text>"); return }
        addTask(os.Args[2])
    case "list":
        listTasks()
    case "done":
        if len(os.Args) < 3 { fmt.Println("❌ Usage: done <id>"); return }
        id, err := strconv.Atoi(os.Args[2])
        if err != nil { fmt.Printf("❌ Invalid ID: %q\n", os.Args[2]); return }
        markDone(id)
    case "delete":
        if len(os.Args) < 3 { fmt.Println("❌ Usage: delete <id>"); return }
        id, err := strconv.Atoi(os.Args[2])
        if err != nil { fmt.Printf("❌ Invalid ID: %q\n", os.Args[2]); return }
        deleteTask(id)
    default:
        fmt.Printf("❌ Unknown command: %q\n\n", command)
        printUsage()
    }
}
