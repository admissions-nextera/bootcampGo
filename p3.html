§1 What You're Building

A URL Shortener API. You give it a long URL, it gives you a short code like go.dev → localhost:8080/r/xK9mP2. Visit the short URL and it redirects you. Track how many times each link was clicked. List all your links. Delete ones you don't want. That's a real product.

§3 Build It — Four Files, One Real Service

01

store.go — The Data Layer

The Link type, the Store struct, and all operations on it. Thread-safe with RWMutex.

Go store.go — annotated
package main
Single package — all files in the project share package main.
import ( "errors" "math/rand" "sync" "time" )
var ErrNotFound = errors.New("link not found")
Exported sentinel error. Handlers check errors.Is(err, ErrNotFound) to decide which HTTP status to return.
type Link struct { Code string `json:"code"` OriginalURL string `json:"original_url"` ShortURL string `json:"short_url"` Clicks int `json:"clicks"` CreatedAt time.Time `json:"created_at"` }
Our core domain type. Every field is exported and tagged for clean JSON output.
type Store struct { mu sync.RWMutex links map[string]Link }
Store wraps the map + mutex together. Unexported fields — nobody outside the package can touch them directly. This is encapsulation the Go way.
func NewStore() *Store { return &Store{links: make(map[string]Link)} }
Constructor function — the Go convention for types that need initialization. Always return a pointer to a struct you want to share across handlers.
func (s *Store) Save(originalURL, baseURL string) Link { code := generateCode(6) link := Link{ Code: code, OriginalURL: originalURL, ShortURL: baseURL + "/r/" + code, CreatedAt: time.Now(), } s.mu.Lock() s.links[code] = link s.mu.Unlock() return link }
Pointer receiver — modifies the Store's data. Lock for writes, unlock immediately after. We return the link so the handler can respond with it.
func (s *Store) Get(code string) (Link, error) { s.mu.RLock() l, ok := s.links[code] s.mu.RUnlock() if !ok { return Link{}, ErrNotFound } return l, nil }
RLock for reads — multiple goroutines can read simultaneously. Only writes need exclusive Lock.
func (s *Store) IncrClick(code string) { s.mu.Lock() defer s.mu.Unlock() if l, ok := s.links[code]; ok { l.Clicks++ s.links[code] = l } }
Maps store copies of structs — you must read, modify, and write back. This is a write operation (incrementing), so it needs a full Lock.
func (s *Store) List() []Link { s.mu.RLock() defer s.mu.RUnlock() links := make([]Link, 0, len(s.links)) for _, l := range s.links { links = append(links, l) } return links }
func (s *Store) Delete(code string) error { s.mu.Lock() defer s.mu.Unlock() if _, ok := s.links[code]; !ok { return ErrNotFound } delete(s.links, code) return nil }
// generateCode returns a random alphanumeric string of length n const charset = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789" func generateCode(n int) string { b := make([]byte, n) for i := range b { b[i] = charset[rand.Intn(len(charset))] } return string(b) }
6-char alphanumeric = 62⁶ = ~56 billion possible codes. More than enough for an in-memory store. Lowercase function = unexported helper.
02

handlers.go — The HTTP Layer

All request handlers. Each one does one job: parse → validate → store → respond.

Go handlers.go — complete 
package main

import (
    "encoding/json"
    "errors"
    "net/http"
    "strings"
)

// Handler holds the store — dependency injection via struct
type Handler struct {
    store *Store
}

func NewHandler(s *Store) *Handler {
    return &Handler{store: s}
}

// writeJSON sets Content-Type and encodes the response
func writeJSON(w http.ResponseWriter, status int, v any) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    if v != nil {
        json.NewEncoder(w).Encode(v)
    }
}

func writeError(w http.ResponseWriter, status int, msg string) {
    writeJSON(w, status, map[string]string{"error": msg})
}

// POST /api/shorten
func (h *Handler) Shorten(w http.ResponseWriter, r *http.Request) {
    var req struct {
        URL string `json:"url"`
    }
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
        writeError(w, http.StatusBadRequest, "invalid JSON")
        return
    }
    if req.URL == "" || !strings.HasPrefix(req.URL, "http") {
        writeError(w, http.StatusBadRequest, "url is required and must start with http")
        return
    }

    baseURL := "http://" + r.Host
    link := h.store.Save(req.URL, baseURL)
    writeJSON(w, http.StatusCreated, link)
}

// GET /api/links
func (h *Handler) ListLinks(w http.ResponseWriter, r *http.Request) {
    writeJSON(w, http.StatusOK, h.store.List())
}

// GET /api/links/{code}
func (h *Handler) GetLink(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    link, err := h.store.Get(code)
    if errors.Is(err, ErrNotFound) {
        writeError(w, http.StatusNotFound, "link not found")
        return
    }
    writeJSON(w, http.StatusOK, link)
}

// DELETE /api/links/{code}
func (h *Handler) DeleteLink(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    err := h.store.Delete(code)
    if errors.Is(err, ErrNotFound) {
        writeError(w, http.StatusNotFound, "link not found")
        return
    }
    writeJSON(w, http.StatusNoContent, nil)
}

// GET /r/{code} — the redirect
func (h *Handler) Redirect(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    link, err := h.store.Get(code)
    if errors.Is(err, ErrNotFound) {
        writeError(w, http.StatusNotFound, "short link not found")
        return
    }
    h.store.IncrClick(code)
    http.Redirect(w, r, link.OriginalURL, http.StatusFound)
}

// GET /health
func healthHandler(w http.ResponseWriter, r *http.Request) {
    writeJSON(w, http.StatusOK, map[string]string{"status": "ok"})
}
03

middleware.go — Logging & CORS

Request logging with duration. CORS headers so a frontend can call this API.

Go middleware.go — complete 
package main

import (
    "fmt"
    "net/http"
    "time"
)

// responseWriter wraps http.ResponseWriter to capture the status code
type responseWriter struct {
    http.ResponseWriter
    status int
}

func (rw *responseWriter) WriteHeader(code int) {
    rw.status = code
    rw.ResponseWriter.WriteHeader(code)
}

// LoggingMiddleware logs method, path, status, and duration for every request
func LoggingMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        rw := &responseWriter{ResponseWriter: w, status: 200}
        next.ServeHTTP(rw, r)
        fmt.Printf("%s  %-7s %-35s → %d  (%dms)\n",
            time.Now().Format("15:04:05"),
            r.Method,
            r.URL.Path,
            rw.status,
            time.Since(start).Milliseconds(),
        )
    })
}

// CORSMiddleware adds headers so browser frontends can call this API
func CORSMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("Access-Control-Allow-Origin", "*")
        w.Header().Set("Access-Control-Allow-Methods", "GET, POST, DELETE, OPTIONS")
        w.Header().Set("Access-Control-Allow-Headers", "Content-Type")
        if r.Method == http.MethodOptions {
            w.WriteHeader(http.StatusNoContent)
            return
        }
        next.ServeHTTP(w, r)
    })
}
🧠
Why we wrap ResponseWriter

The standard http.ResponseWriter gives you no way to read back the status code after writing it — it's write-only. The responseWriter wrapper intercepts WriteHeader() calls and stores the code, letting our logger report the actual HTTP status. This is a classic Go middleware pattern you'll see in every real codebase.

04

main.go — Wire It All Together

Create the store, create the handler, register routes, chain middleware, start the server.

Go main.go — complete 
package main

import (
    "fmt"
    "log"
    "net/http"
)

func main() {
    // Dependency injection — create store once, share via Handler
    store   := NewStore()
    handler := NewHandler(store)
    mux     := http.NewServeMux()

    // API routes
    mux.HandleFunc("POST /api/shorten",         handler.Shorten)
    mux.HandleFunc("GET /api/links",             handler.ListLinks)
    mux.HandleFunc("GET /api/links/{code}",       handler.GetLink)
    mux.HandleFunc("DELETE /api/links/{code}",    handler.DeleteLink)

    // Redirect route
    mux.HandleFunc("GET /r/{code}", handler.Redirect)

    // Health check
    mux.HandleFunc("GET /health", healthHandler)

    // Chain middleware: request flows Logging → CORS → mux → handler
    app := LoggingMiddleware(CORSMiddleware(mux))

    port := ":8080"
    fmt.Printf("🔗 URL Shortener running on http://localhost%s\n", port)
    fmt.Println("   POST /api/shorten     → create short link")
    fmt.Println("   GET  /api/links       → list all links")
    fmt.Println("   GET  /r/{code}        → redirect")
    fmt.Println()

    if err := http.ListenAndServe(port, app); err != nil {
        log.Fatalf("server failed: %v", err)
    }
}

§5 How to Keep Growing — Extension Challenges

⚡
Challenge 1 — Custom codes

Let users pass "custom_code": "mylink" in the POST body. If the code is already taken, return 409 Conflict. If not provided, generate one as usual.

⚡
Challenge 2 — Expiry

Add an ExpiresAt *time.Time field to Link. Accept an optional "ttl_hours" in the POST body. In the Redirect handler, check if the link has expired and return 410 Gone if so.

⚡
Challenge 3 — Persist to disk

When the server starts, load existing links from a links.json file if it exists. On every write (Save, Delete), marshal the entire store and write it to links.json. Now your data survives restarts.

🏆
Challenge 4 — Write tests

Create store_test.go and handlers_test.go. Use Go's built-in testing package and net/http/httptest to test your handlers without running a real server. This is how professional Go code is tested.
